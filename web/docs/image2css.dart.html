readFile(ArrayBuffer buffer){
  print("Reading...");
  var array = new Uint8Array.fromBuffer(buffer);
  var arrayLength = array.length;
  var width = _readInt(array.getRange(18, 4));
  var height = _readInt(array.getRange(22, 4));
  var current = 54;
  var outBuffer = new StringBuffer();
  for(int y = height - 1; y >= 0; y--){
    for(int x = 0; x < width; x++){
      var color = _readColor(array.getRange(current, 3));
      current+=3;
      outBuffer.add("${x*5}px ${y*5}px 5px 5px $color");
      outBuffer.add( (current == arrayLength) ? ";" : ",");
      outBuffer.add("$current $arrayLength\n");
    }
  }
  var style = query("#cssImage").style;
  style.backgroundColor = "red";
  //query("#cssImage").style.boxShadow = "0px 5px 5px 5px #ffaec9;";
  //outBuffer.toString();
  print(outBuffer.toString());
  //query("#cssImage").classes["cssImage"] style.position = "relative";
}
